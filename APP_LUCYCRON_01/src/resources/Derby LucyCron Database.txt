1. 필요한 환경변수

		DERBY_OPTS=-Dderby.system.home=N:/tain/products/LucyCron/data/derbyDB   <- derby database

		> java -Dderby.system.home=%DB_HOME%/data/derbyDB



2. 처음 Embedded Database

		DOS> ij
		ij 버전 10.8
		
		ij> connect 'jdbc:derby:firstdb;create=true';
		
		ij> create table firsttable
		> (
		>   id int primary key,
		>   name varchar(12)
		> );
		
		ij> insert into firsttable values
		> (10, 'ten'),
		> ( 11, 'eleven'),
		> (30, 'thirty');
		
		ij> run 'loadTable.sql';
		
		ij> disconnect;
		
		ij> exit;
		
		DOS> dir
		
			firstdb

4. derby.properties

	set DERBY_OPTS=-Dderby.system.home=%TAIN_HOME%\data\derbyDB
	-Dderby.system.home=%TAIN_HOME%\data\derbyDB 위치에 아래와 같은 내용으로 생성한다.
	
	DOS> type derby.properties
	
		# 호스트정보
		derby.drda.host=127.0.0.1
		# 서버 포트 설정
		derby.drda.portNumber=1110

		# 쓰레드 최대 수
		derby.drda.maxThreads=50
		# 쓰레드 최소 수
		derby.drda.minThreads=10
		# 계정 인증 여부

		derby.connection.requireAuthentication=true
		# 인증 방식
		derby.authentication.provider=BUILTIN
		# user.[계정]=[비밀번호]
		derby.user.kang=kang123!
